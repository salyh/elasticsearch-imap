026package de.saly.elasticsearch.riverstate;
027
028import java.util.Date;
029
030public class RiverState {
031
032    private boolean exists;
033
034    private String folderUrl;
035
036    private long lastCount = -1;
037
038    private Date lastIndexed;
039
040    private Date lastSchedule;
041
042    private long lastTook = -1;
043
044    private long lastUid = -1;
045
046    private Long uidValidity;
047
048    @Override
049    public boolean equals(final Object obj) {
050        if (this == obj) {
051            return true;
052        }
053        if (obj == null) {
054            return false;
055        }
056        if (getClass() != obj.getClass()) {
057            return false;
058        }
059        final RiverState other = (RiverState) obj;
060        if (folderUrl == null) {
061            if (other.folderUrl != null) {
062                return false;
063            }
064        } else if (!folderUrl.equals(other.folderUrl)) {
065            return false;
066        }
067        return true;
068    }
069
070    public String getFolderUrl() {
071        return folderUrl;
072    }
073
074    public long getLastCount() {
075        return lastCount;
076    }
077
078    public Date getLastIndexed() {
079        return lastIndexed;
080    }
081
082    public Date getLastSchedule() {
083        return lastSchedule;
084    }
085
086    public long getLastTook() {
087        return lastTook;
088    }
089
090    public long getLastUid() {
091        return lastUid;
092    }
093
094    public Long getUidValidity() {
095        return uidValidity;
096    }
097
098    @Override
099    public int hashCode() {
100        final int prime = 31;
101        int result = 1;
102        result = prime * result + (folderUrl == null ? 0 : folderUrl.hashCode());
103        return result;
104    }
105
106    public boolean isExists() {
107        return exists;
108    }
109
110    public void setExists(final boolean exists) {
111        this.exists = exists;
112    }
113
114    public void setFolderUrl(final String folderUrl) {
115        this.folderUrl = folderUrl;
116    }
117
118    public void setLastCount(final long lastCount) {
119        this.lastCount = lastCount;
120    }
121
122    public void setLastIndexed(final Date lastIndexed) {
123        this.lastIndexed = lastIndexed;
124    }
125
126    public void setLastSchedule(final Date lastSchedule) {
127        this.lastSchedule = lastSchedule;
128    }
129
130    public void setLastTook(final long lastTook) {
131        this.lastTook = lastTook;
132    }
133
134    public void setLastUid(final long lastUid) {
135        this.lastUid = lastUid;
136    }
137
138    public void setUidValidity(final Long uidValidity) {
139        this.uidValidity = uidValidity;
140    }
141
142    @Override
143    public String toString() {
144        return "RiverState [folderUrl=" + folderUrl + ", uidValidity=" + uidValidity + ", lastUid=" + lastUid + ", lastSchedule="
145                + lastSchedule + ", lastIndexed=" + lastIndexed + ", lastTook=" + lastTook + ", exists=" + exists + ", lastCount="
146                + lastCount + "]";
147    }
148
149}
