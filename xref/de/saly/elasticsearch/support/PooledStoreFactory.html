26  package de.saly.elasticsearch.support;
27  
28  import java.util.Properties;
29  
30  import javax.mail.Session;
31  import javax.mail.Store;
32  
33  import org.apache.commons.pool2.BasePooledObjectFactory;
34  import org.apache.commons.pool2.PooledObject;
35  import org.apache.commons.pool2.impl.DefaultPooledObject;
36  
37  public class PooledStoreFactory extends BasePooledObjectFactory<Store> {
38  
39      private final String password;
40      private final Properties props;
41      private final String user;
42  
43      public PooledStoreFactory(final Properties props, final String user, final String password) {
44          super();
45          this.props = props;
46          this.user = user;
47          this.password = password;
48      }
49  
50      @Override
51      public Store create() throws Exception {
52          final Session session = Session.getInstance(props, null);
53          final Store store = session.getStore();
54          store.connect(user, password);
55          return store;
56      }
57  
58      @Override
59      public void destroyObject(final PooledObject<Store> p) throws Exception {
60          final Store store = p.getObject();
61          IMAPUtils.close(store);
62      }
63  
64      @Override
65      public boolean validateObject(final PooledObject<Store> p) {
66          final Store store = p.getObject();
67          return store != null && store.isConnected();
68      }
69  
70      @Override
71      public PooledObject<Store> wrap(final Store obj) {
72          return new DefaultPooledObject<Store>(obj);
73      }
74  
75  }
